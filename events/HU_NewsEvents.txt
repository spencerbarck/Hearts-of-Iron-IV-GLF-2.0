# Great Lakes Confederation Joins Allies
news_event = {
	id = hu_news.2
	title = hu_news.2.t
	picture = GFX_news_event_churchill_degaulle
	desc = hu_news.2.d

	major = yes

	is_triggered_only = yes

	#We stand together.
	option = {
		name = hu_news.2.a
		trigger = {
			OR = {
				tag = ENG
				tag = GLF
			}
		}
	}

	#They are preparing for the worst.
	option = {
		name = hu_news.2.b
		trigger = {
			GLF = {
				has_war = no
			}
			NOT = {
				OR = {
					tag = GLF
					tag = ENG
					tag = GER
				}
			}
		}
	}
	#Banding together like cornered rats.
	option = {
		name = hu_news.2.c
		trigger = {
			OR = {
				tag = GER
				has_war_with = GLF
			}
		}
	}

	#Britain has thrown in their lot.
	option = {
		name = hu_news.2.e
		trigger = {
			GLF = {
				has_war = yes
			}
			NOT = {
				has_war_with = GLF
			}
			NOT = {
				OR = {
					tag = GLF
					tag = ENG
					tag = GER
				}
			}
		}
	}
}
